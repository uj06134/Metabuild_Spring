drop table football;
create table football(
num number primary key,
id varchar2(10),
pw varchar2(20),
win varchar2(10),
round16 varchar2(60)
);

drop sequence fb_seq;
create sequence fb_seq
minvalue 1
start with 1
nocache;

commit;

col id for a6
col pw for a6
col win for a10
col round16 for a30
select * from football;
